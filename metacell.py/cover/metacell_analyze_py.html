""" 

Analyze a profiles container. 

""" 

 

from metacell.parameters import load_profiles 

from metacell.storage.profiles import Profiles 

from metacell.storage.profiles import ProfilesCollection 

from metacell.storage.profiles import ProfilesGroups 

from metacell.storage.profiles import ProfilesView 

from tgutils.application import * # pylint: disable=wildcard-import,unused-wildcard-import 

 

import metacell.utilities as ut 

import tgutils.numpy as np 

 

 

@config(top=True) 

def total_umis() -> None: 

""" 

Compute and store the total UMIs per profile and per gene. 

""" 

profiles = load_profiles() 

 

if isinstance(profiles, ProfilesCollection): 

_compute_total_collection_umis(profiles) 

return 

 

if isinstance(profiles, ProfilesView) and not profiles.is_slicing_profiles(): 

if not profiles.is_slicing_genes(): 

return 

 

base_genes_total_umis = profiles.base_profiles.g_array(np.ArrayFloat32, 'total_umis') 

view_genes_total_umis = base_genes_total_umis[profiles.base_gene_indices] 

 

if view_genes_total_umis.sum() == base_genes_total_umis.sum(): 

profiles_total_umis = profiles.base_profiles.p_array(np.ArrayFloat32, 'total_umis') 

np.ArrayFloat32.write(profiles_total_umis, profiles.data_path('p.total_umis')) 

np.ArrayFloat32.write(view_genes_total_umis, profiles.data_path('g.total_umis')) 

return 

 

_compute_total_container_umis(profiles) 

 

 

def _compute_total_collection_umis(profiles: ProfilesCollection) -> None: 

genes_total_umis = np.ArrayFloat32.zeros(profiles.genes.count) 

for container in profiles.containers: 

genes_total_umis += container.g_array(np.ArrayFloat32, 'total_umis') # type: ignore 

 

np.ArrayFloat32.write(genes_total_umis, profiles.data_path('g.total_umis')) 

 

 

def _compute_total_container_umis(profiles: Profiles) -> None: 

processes_count = processes() 

profiles_total_umis = np.ArrayFloat32.shared_memory_zeros(profiles.profiles_count) 

processes_genes_total_umis = \ 

np.MatrixFloat32.shared_memory_zeros((processes_count, profiles.genes.count)) 

 

with ut.sum_profiles_loop(profiles.profiles_count) as loop: 

parallel(processes_count, 

_sum_total_profiles_umis, 

profiles, 

profiles_total_umis, 

loop, 

kwargs=lambda index: dict( # 

profiles_range=indexed_range(index, size=profiles.profiles_count, 

invocations=processes_count), 

genes_total_umis=processes_genes_total_umis[index, :])) 

 

genes_total_umis = np.ArrayFloat32.be(processes_genes_total_umis.sum(axis=0)) 

assert len(genes_total_umis) == profiles.genes.count 

 

np.ArrayFloat32.write(genes_total_umis, profiles.data_path('g.total_umis')) 

np.ArrayFloat32.write(profiles_total_umis, profiles.data_path('p.total_umis')) 

 

 

def _sum_total_profiles_umis(profiles: Profiles, profiles_total_umis: np.ArrayFloat32, loop: Loop, 

*, profiles_range: range, genes_total_umis: np.ArrayFloat32) -> None: 

profile_umis = np.ArrayFloat32.empty(profiles.genes.count) 

 

for profile_index in profiles_range: 

profile_umis.fill(0) 

profiles.add_into_gp_array(np.ArrayFloat32, 'umis', profile_index, profile_umis) 

genes_total_umis += profile_umis # type: ignore 

profiles_total_umis[profile_index] = profile_umis.sum() 

loop.step() 

 

 

@config(top=True) 

def collect_groups() -> None: 

""" 

Create a profiles container where each profile is the sum of the grouped profiles. 

""" 

profiles = load_profiles() 

 

groups_dir = os.path.join(profiles.metadata.path, 'groups') 

os.makedirs(groups_dir, exist_ok=True) 

 

groups_of_profiles = profiles.p_array(np.ArrayInt32, 'group_index') 

groups_count = groups_of_profiles.max() + 1 

profiles_count = (groups_count >= 0).sum() 

 

groups_umis_matrix = np.MatrixFloat32.shared_memory_zeros((profiles.genes.count, groups_count)) 

 

with ut.sum_profiles_loop(profiles_count) as loop: 

grouped_count = \ 

parallel(groups_count, 

_sum_group_profiles, 

profiles, 

groups_of_profiles, 

loop, 

kwargs=lambda index: dict(group_index=index, 

group_umis=groups_umis_matrix[:, index])) 

 

np.MatrixFloat32.write(groups_umis_matrix, os.path.join(groups_dir, 'gp.umis')) 

np.ArrayInt32.write(np.ArrayInt32.be(grouped_count), 

os.path.join(groups_dir, 'p.grouped_count')) 

 

ProfilesGroups.created(groups_dir, profiles) 

 

 

def _sum_group_profiles(profiles: Profiles, groups_of_profiles: np.ArrayInt32, 

loop: Loop, *, group_index: int, group_umis: np.ArrayFloat32) -> int: 

profile_indices = np.where(groups_of_profiles == group_index)[0] 

for profile_index in profile_indices: 

profiles.add_into_gp_array(np.ArrayFloat32, 'umis', profile_index, group_umis) 

loop.step() 

 

return len(profile_indices)
