""" 

Build steps for importing raw data into the metacell pipeline. 

""" 

 

from metacell.dynamake.common import call 

from metacell.dynamake.common import ecall 

from tgutils.make import * # pylint: disable=wildcard-import,unused-wildcard-import 

 

import os 

 

# pylint: disable=missing-docstring 

 

 

@step(output=phony('std')) 

async def make_std() -> None: 

require('std/.done') 

 

 

@step(output=phony('delaned')) 

async def make_delaned() -> None: 

require('delaned/.done') 

 

 

@step(output=phony('consistent_genes')) 

async def make_consistent_genes() -> None: 

require('delaned/.consistent_genes') 

 

 

@step(output=phony('genes')) 

async def make_genes() -> None: 

require('genes/genes.yaml') 

 

 

@step(output=phony('cells')) 

async def make_cells() -> None: 

require('cells/profiles.yaml') 

 

 

@step(output='std/.done') 

async def convert_all() -> None: 

tg_require(glob_fmt(optional('raw/{*name}/matrix.mtx'), 'std/{name}/format.yaml')) # 10X 

tg_require(glob_fmt(optional('raw/{*name}/gene_count.txt'), 'std/{name}/format.yaml')) # MOCA 

tg_require(glob_fmt(optional('raw/{*name}.h5'), 'std/{name}/format.yaml')) # HCA 

await shell('touch', emphasized('std/.done')) 

 

 

@step(output=[exists('std/{*name}/{**_lane}/umis.mtx', 

'std/{*name}/{**_lane}/profiles.csv', 

'std/{*name}/{**_lane}/genes.csv'), 

'std/{*name}/{**_lane}/format.yaml']) 

async def convert_dir(**_kwargs: str) -> None: 

if Stat.exists(e('raw/{name}/matrix.mtx')): 

await done(convert_10x()) 

return 

 

if Stat.exists(e('raw/{name}/gene_count.txt')): 

await done(convert_moca()) 

return 

 

if Stat.exists(e('raw/{name}.h5')): 

await done(convert_hca()) 

return 

 

raise RuntimeError(e('unknown raw data type: raw/{name}')) 

 

 

async def convert_10x() -> None: 

erequire('raw/{name}/matrix.mtx', 'raw/{name}/barcodes.tsv', 'raw/{name}/genes.tsv') 

 

await eshell('@rm -rf std/{name}') 

await eshell('@mkdir -p std/{name}') 

 

await eshell('cp $PWD/raw/{name}/matrix.mtx', emphasized('std/{name}/umis.mtx')) 

 

await eshell('sed "/-/!s/$/,/;s/-/,/" $PWD/raw/{name}/barcodes.tsv >', 

emphasized('std/{name}/profiles.csv')) 

 

await eshell('cp $PWD/raw/{name}/genes.tsv', emphasized('std/{name}/genes.csv')) 

 

await eshell('cp', standard_format_file('10x'), emphasized('std/{name}/format.yaml')) 

 

 

async def convert_moca() -> None: 

erequire('raw/{name}/gene_count.txt', 

'raw/{name}/cell_annotate.csv', 

'raw/{name}/gene_annotate.csv') 

 

await eshell('@rm -rf std/{name}') 

await eshell('@mkdir -p std/{name}') 

 

await eshell('cp $PWD/raw/{name}/gene_count.txt', emphasized('std/{name}/umis.mtx')) 

 

await eshell('sed "s/sample,/lane,barcode/;s/\\./,/" $PWD/raw/{name}/cell_annotate.csv >', 

emphasized('std/{name}/profiles.csv')) 

 

await eshell('sed "s/gene_id/ensembl/;s/gene_type/type/;s/gene_name/name/"', 

'$PWD/raw/{name}/gene_annotate.csv >', 

emphasized('std/{name}/genes.csv')) 

 

await eshell('cp', standard_format_file('moca'), emphasized('std/{name}/format.yaml')) 

 

 

async def convert_hca() -> None: 

h5_path = e('raw/{name}.h5') 

erequire(h5_path) 

 

await eshell('@rm -rf std/{name}') 

 

await ecall('convert_hca', 

std_dir='std/{name}', 

raw_file=h5_path) 

 

await eshell('cp', standard_format_file('hca'), emphasized('std/{name}/format.yaml')) 

 

 

@step(output='delaned/.done') 

async def delane_all() -> None: 

require('std/.done') 

await sync() 

tg_require(glob_fmt('std/{*name}/format.yaml', 'delaned/{name}/.done')) 

await shell('touch', emphasized('delaned/.done')) 

 

 

@step(output=['delaned/{*name}/.done', 

exists('delaned/{*name}/{**_lane}/umis.mtx', 

'delaned/{*name}/{**_lane}/profiles.csv', 

'delaned/{*name}/{**_lane}/genes.csv', 

'delaned/{*name}/{**_lane}/format.yaml')]) 

async def delane_dir(**_kwargs: str) -> None: 

erequire('std/{name}/format.yaml') 

 

await eshell('@rm -rf delaned/{name}') 

 

await ecall('delane_dir', 

delaned_root='delaned/{name}', 

std_dir='std/{name}') 

 

await eshell('touch', emphasized('delaned/{name}/.done')) 

 

 

@step(output=['delaned/.consistent_genes']) 

async def consistent_genes() -> None: 

require('delaned/.done') 

await sync() 

captured = glob_capture('delaned/{**name}/genes.csv') 

context()['first_genes_path'] = captured.paths[0] 

tg_require(fmts(captured.wildcards[1:], 'delaned/{name}/.consistent_lane_genes')) 

await eshell('touch', emphasized('delaned/.consistent_genes')) 

 

 

@step(output='delaned/{**lane}/.consistent_lane_genes') 

async def consistent_lane_genes(**_kwargs: str) -> None: 

first_genes_path = context()['first_genes_path'] 

await eshell('cmp ', first_genes_path, 'delaned/{lane}/genes.csv', 

'|| diff', first_genes_path, emphasized('delaned/{lane}/genes.csv')) 

await eshell('touch', emphasized('delaned/{lane}/.consistent_lane_genes')) 

 

 

@step(output=['genes/genes.yaml', 

optional('genes/g.{*_field}.txt', 

'genes/g.{*_field}.npy')]) 

async def import_genes(**_kwargs: str) -> None: 

require('delaned/.consistent_genes') 

await sync() 

genes_paths = sorted(glob_paths('delaned/{**name}/genes.csv')) 

first_std_dir = os.path.dirname(genes_paths[0]) 

await shell('@rm -rf genes') 

await call('import_genes', 

genes_dir='genes', 

std_dir=first_std_dir, 

organism=config_param('organism', is_required=True)) 

 

 

@step(output='cells/profiles.yaml') 

async def collect_cells() -> None: 

require('cells/.imported') 

await sync() 

batch_dirs = glob_fmt('cells/{*name}/{**lane}/profiles.yaml', '{name}/{lane}') 

await call('collect_profiles', 

collection_dir='cells', 

combined_dirs=' '.join(batch_dirs)) 

 

 

@step(output='cells/.imported') 

async def import_cells() -> None: 

require('delaned/.done') 

await sync() 

captured = glob_capture('delaned/{*name}/{**lane}/umis.mtx') 

batch_paths = fmts(captured.wildcards, 'cells/{name}/{lane}/profiles.yaml') 

tg_require(batch_paths) 

await eshell('touch', emphasized('cells/.imported')) 

 

 

@step(output=['cells/{*name}/{**lane}/profiles.yaml', 

'cells/{*name}/{**lane}/gp.umis.mmm', 

'cells/{*name}/{**lane}/p.barcode.txt', 

'cells/{*name}/{**lane}/p.batch.txt', 

'cells/{*name}/{**lane}/p.lane.txt']) 

async def import_batch(**_kwargs: str) -> None: 

require('genes') 

erequire('delaned/{name}/.done') 

await eshell('@rm -rf cells/{name}/{lane}') 

 

await ecall('import_profiles', 

batch_dir='cells/{name}/{lane}', 

std_dir='delaned/{name}/{lane}', 

batch_name='{name}') 

 

await ecall('import_umis', 

batch_dir='cells/{name}/{lane}', 

std_dir='delaned/{name}/{lane}', 

genes_dir='genes') 

 

await ecall('imported_batch', 

batch_dir='cells/{name}/{lane}', 

genes_dir='genes', 

profiles_kind=config_param('profiles_kind', 'cells')) 

 

 

def standard_format_file(name: str) -> str: 

return os.path.join(os.path.dirname(os.path.dirname(__file__)), 

'storage', '%s_format.yaml' % name)
